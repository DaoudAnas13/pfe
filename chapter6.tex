\chapter{Sprint 3: File Management and Geolocation Tracking}

\section{Introduction}

Sprint 3 extends the core intervention management system with file management capabilities and geolocation tracking. This sprint enables technicians to capture and upload photographic evidence, attach documents, and track intervention location data. The implementation provides secure file storage, offline file queuing for mobile devices, and location-aware intervention tracking.

The sprint delivers critical functionality for evidence management, allowing supervisors and clients to access comprehensive documentation of completed interventions. Geolocation tracking provides accountability and operational insights into technician activities and intervention execution locations.

\textbf{File Upload Service:}
\begin{lstlisting}
@Service
public class FileUploadService {

    private final FilesRepository filesRepository;
    private final FileStorageService storageService;
    private final ImageCompressionService compressionService;

    public FileDTO uploadFile(MultipartFile file, Long requestId) {
        // Validate file
        validateFileUpload(file);

        // Compress if image
        MultipartFile processedFile = file;
        if (isImageFile(file)) {
            processedFile = compressionService.compress(file);
        }

        // Store file
        String filePath = storageService.saveFile(processedFile);

        // Create file record
        Files fileEntity = new Files();
        fileEntity.setFilename(file.getOriginalFilename());
        fileEntity.setFileType(file.getContentType());
        fileEntity.setFileSize(processedFile.getSize());
        fileEntity.setFilePath(filePath);
        fileEntity.setUploadDate(ZonedDateTime.now());

        RapportRequest request = requestRepository.findById(requestId)
            .orElseThrow();
        fileEntity.setRapportRequest(request);

        fileEntity = filesRepository.save(fileEntity);
        return mapper.toDto(fileEntity);
    }

    private void validateFileUpload(MultipartFile file) {
        if (file.isEmpty()) {
            throw new IllegalArgumentException("File is empty");
        }
        if (file.getSize() > 50 * 1024 * 1024) { // 50MB limit
            throw new IllegalArgumentException("File too large");
        }
    }

    private boolean isImageFile(MultipartFile file) {
        String contentType = file.getContentType();
        return contentType != null && contentType.startsWith("image/");
    }
}
\end{lstlisting}

class FileUploadBloc extends Bloc<FileUploadEvent, FileUploadState> {
  final FileUploadRepository fileRepository;
  final ImageCompressionService compressionService;

  FileUploadBloc(this.fileRepository, this.compressionService)
      : super(FileUploadInitial()) {
    on<UploadFile>(_onUploadFile);
    on<QueueFileOffline>(_onQueueFileOffline);
  }

  Future<void> _onUploadFile(
    UploadFile event,
    Emitter<FileUploadState> emit,
  ) async {
    emit(FileUploading());
    try {
      // Compress image if needed
      File processedFile = event.file;
      if (_isImageFile(event.file)) {
        processedFile = await compressionService.compressImage(event.file);
      }

      // Upload file
      final response = await fileRepository.uploadFile(
        processedFile,
        event.requestId,
      );

      emit(FileUploadSuccess(file: response));
    } on DioError catch (e) {
      if (e.type == DioErrorType.other) {
        // Queue offline
        await _queueOffline(event.file, event.requestId);
        emit(FileQueuedOffline());
      } else {
        emit(FileUploadFailure(error: e.message ?? 'Upload failed'));
      }
    }
  }

  bool _isImageFile(File file) {
    final ext = file.path.split('.').last.toLowerCase();
    return ['jpg', 'jpeg', 'png', 'gif'].contains(ext);
  }

  Future<void> _queueOffline(File file, int requestId) async {
    // Store in local database for later sync
  }
}

\textbf{Geolocation Service:}
\begin{lstlisting}
class GeolocationService {
  final GeolocatorPlatform _geolocatorPlatform = GeolocatorPlatform.instance;

  Future<Position?> getCurrentLocation() async {
    try {
      final hasPermission = await _handleLocationPermission();
      if (!hasPermission) {
        return null;
      }

      const LocationSettings locationSettings = LocationSettings(
        accuracy: LocationAccuracy.best,
        distanceFilter: 5, // 5 meters
      );

      return await _geolocatorPlatform.getCurrentPosition(
        locationSettings: locationSettings,
      );
    } catch (e) {
      print('Error getting location: $e');
      return null;
    }
  }

  Future<bool> _handleLocationPermission() async {
    bool serviceEnabled = await _geolocatorPlatform.isLocationServiceEnabled();
    if (!serviceEnabled) {
      return false;
    }

    LocationPermission permission = await Geolocator.checkPermission();
    if (permission == LocationPermission.denied) {
      permission = await Geolocator.requestPermission();
    }

    return permission != LocationPermission.denied &&
        permission != LocationPermission.deniedForever;
  }
}
