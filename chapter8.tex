\textbf{Excel Report Service:}
\begin{lstlisting}
@Service
public class ExcelReportService {

    private final RapportRequestRepository requestRepository;
    private final ConfigExportRepository configRepository;
    private final FileStorageService storageService;

    public ResponseEntity<InputStreamResource> generateReport(
        Long configId,
        Map<String, Object> filters) throws IOException {

        ConfigExport config = configRepository.findById(configId)
            .orElseThrow();

        // Fetch data
        List<RapportRequest> data = requestRepository
            .findAll(buildSpecification(filters));

        // Create workbook
        Workbook workbook = new XSSFWorkbook();
        Sheet sheet = workbook.createSheet("Interventions");

        // Add headers
        addHeaders(sheet, config);

        // Add data
        addData(sheet, data, config);

        // Create file
        File tempFile = File.createTempFile("report_", ".xlsx");
        try (FileOutputStream fos = new FileOutputStream(tempFile)) {
            workbook.write(fos);
        }

        // Prepare download
        InputStreamResource resource = new InputStreamResource(
            Files.newInputStream(tempFile.toPath()));

        return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_DISPOSITION,
                "attachment; filename=\"rapport_" + 
                LocalDate.now() + ".xlsx\"")
            .contentType(MediaType.APPLICATION_OCTET_STREAM)
            .body(resource);
    }

    private void addHeaders(Sheet sheet, ConfigExport config) {
        Row headerRow = sheet.createRow(0);
        int cellNum = 0;

        for (Map.Entry<String, String> entry : 
            config.getColumns().entrySet()) {
            Cell cell = headerRow.createCell(cellNum++);
            cell.setCellValue(entry.getValue());
            cell.setCellStyle(getHeaderStyle(sheet.getWorkbook()));
        }
    }

    private void addData(Sheet sheet, List<RapportRequest> data,
        ConfigExport config) {

        int rowNum = 1;
        for (RapportRequest request : data) {
            Row row = sheet.createRow(rowNum++);
            int cellNum = 0;

            for (String fieldName : config.getColumns().keySet()) {
                Cell cell = row.createCell(cellNum++);
                Object value = extractFieldValue(request, fieldName);
                setCellValue(cell, value);
            }
        }
    }

    private CellStyle getHeaderStyle(Workbook workbook) {
        CellStyle style = workbook.createCellStyle();
        Font font = workbook.createFont();
        font.setBold(true);
        style.setFont(font);
        style.setFillForegroundColor(IndexedColors.LIGHT_BLUE.getIndex());
        style.setFillPattern(FillPatternType.SOLID_FOREGROUND);
        return style;
    }

    private Object extractFieldValue(RapportRequest request,
        String fieldName) {

        switch (fieldName) {
            case "reference":
                return request.getReference();
            case "status":
                return request.getStatus().getLabel();
            case "createdDate":
                return request.getCreatedDate();
            case "executionDate":
                return request.getExecutionDetails().getStartTime();
            default:
                return "";
        }
    }
}

@Service
public class ArchiveService {

    private final RapportRequestRepository requestRepository;
    private final ArchiveLogRepository archiveLogRepository;

    @Scheduled(cron = "0 2 * * *") // Run daily at 2 AM
    public void archiveOldInterventions() {
        ZonedDateTime cutoffDate = ZonedDateTime.now()
            .minusYears(1);

        List<RapportRequest> oldRequests = requestRepository
            .findByStatusAndCreatedDateBefore(
                RapportStatus.ARCHIVED,
                cutoffDate);

        if (oldRequests.isEmpty()) {
            return;
        }

        // Archive to cold storage
        archiveToStorage(oldRequests);

        // Delete from active database
        requestRepository.deleteAll(oldRequests);

        // Log archive operation
        ArchiveLog log = new ArchiveLog();
        log.setArchivedDate(ZonedDateTime.now());
        log.setRecordCount((long) oldRequests.size());

\textbf{REST Controller:}
\begin{lstlisting}
@RestController
@RequestMapping("/api/reports")
public class ReportController {

    private final ExcelReportService reportService;
    private final ConfigExportRepository configRepository;

    @GetMapping("/generate/{configId}")
    public ResponseEntity<InputStreamResource> generateReport(
        @PathVariable Long configId,
        @RequestParam(required = false) String status,
        @RequestParam(required = false) String startDate,
        @RequestParam(required = false) String endDate) throws IOException {

        Map<String, Object> filters = new HashMap<>();
        if (status != null) filters.put("status", status);
        if (startDate != null) filters.put("startDate", startDate);
        if (endDate != null) filters.put("endDate", endDate);

        return reportService.generateReport(configId, filters);
    }

    @GetMapping("/configs")
    public ResponseEntity<List<ConfigExportDTO>> getReportConfigs() {
        List<ConfigExport> configs = configRepository.findByIsActive(true);
        return ResponseEntity.ok(configs.stream()
            .map(mapper::toDto)
            .collect(Collectors.toList()));
    }

    @PostMapping("/configs")
    @PreAuthorize("hasRole('ADMIN')")
    public ResponseEntity<ConfigExportDTO> createConfig(
        @RequestBody @Valid ConfigExportDTO configDTO) {

        ConfigExport config = mapper.toEntity(configDTO);
        config = configRepository.save(config);
        return ResponseEntity.ok(mapper.toDto(config));
    }
}

@Repository
public interface RapportRequestRepository
    extends JpaRepository<RapportRequest, Long>,
            JpaSpecificationExecutor<RapportRequest> {

    @Query("SELECT r FROM RapportRequest r " +
           "LEFT JOIN FETCH r.template " +
           "LEFT JOIN FETCH r.files " +
           "LEFT JOIN FETCH r.executionDetails " +
           "WHERE r.status = :status")
    List<RapportRequest> findByStatusWithEager(
        @Param("status") RapportStatus status);

    @Cacheable("rapportRequests")
